CREATE TABLE escolaridades (
    id SERIAL PRIMARY KEY,
    descricao VARCHAR(100) UNIQUE
);

INSERT INTO escolaridades(descricao)
SELECT DISTINCT escolaridade
FROM pessoas;

ALTER TABLE pessoas
ADD COLUMN escolaridade_id INT;

UPDATE pessoas p
SET escolaridade_id = e.id
FROM escolaridades e
WHERE p.escolaridade = e.descricao;

ALTER TABLE pessoas
ADD CONSTRAINT fk_pessoas_escolaridade
FOREIGN KEY (escolaridade_id)
REFERENCES escolaridades(id);

ALTER TABLE pessoas
DROP COLUMN escolaridade;
